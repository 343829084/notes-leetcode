# Add Two Numbers

## 题目

给定两个用链表表示的非负数（个位-十位-百位-……），返回表示两数之和的链表

```
Input: (2 -> 4 -> 3) + (5 -> 6 -> 4)
Output: 7 -> 0 -> 8
```

## 思路

计算时，需要考虑进位因素。

## 代码

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
        ListNode *ans = new ListNode(0);
        int carry = 0, n;
        ListNode *cur = ans;
        while (l1 || l2 || carry) {
            n = carry;
            if (l1) {
                n += l1 -> val;
                l1 = l1 -> next;
            }
            if (l2) {
                n += l2 -> val;
                l2 = l2 -> next;
            }
            carry = n / 10;
            n %= 10;
            cur -> next = new ListNode(n);
            cur = cur -> next;
        }
        return ans -> next;
    }
};
```
